230 static hsm_state RunGenericEvent( hsm_status status )
231 {
232  hsm_state next_state=NULL;
233  state_t* state= StateFromStatus( status );
234  const process_t* et;
235  // look through the specified event processors
236  for (et= state->process; et && !next_state; et=et->next) {
237  if (et->flags & ProcessCallback) {
238  next_state= CALL_PROCESS( et, status );
239  }
240  // this block of code could have been installed as a custom process callback
241  // but this avoids a separate function call, and is pretty straight forward
242  // dont know, maybe i will revist for improved code aethetics
243  else {
244  handler_t * handler= (handler_t*)et;
245  // determine if some guard blocks the event from running
246  guard_t* guard;
247  for (guard= handler->guard; guard; guard=guard->next) {
248  if (!CALL_GUARD( guard, status )) {
249  break;
250  }
251  }
252  // no guard blocks this handler from running,:
253  if (!guard) {
254  // run action(s)
255  action_t* at;
256  for (at= handler->actions; at; at=at->next) {
257  at->run( status, at->action_data );
258  }
259  // transition to target, or flag as handled.
260  if (handler->target) {
261  next_state= (hsm_state) handler->target->clientData;
262  }
263  else {
264  next_state= HsmStateHandled();
265  }
266  }
267  }
268  }
269  return next_state;
270 }

697 static hsm_state BuildingStateEvent( hsm_status status )
698 {
699  hsm_state ret= NULL;
700  builder_t * builder= ((builder_t*)status->ctx);
701  state_t* current= Builder_CurrentState( builder );
702  switch (status->evt->type)
703  {

714  ret= HsmStateError();
715  }
716  }
717  break;
718  case _hsm_enter_raw: {
719  const RawEnterEvent* event= (const RawEnterEvent*)status->evt;
720  if (!current->desc.enter && event->enter) {
721  current->desc.enter= event->enter;
722  ret= BuildingBody();
723  }
724  else {
725  Builder_Error( builder, current->desc.enter ? "enter already specified." : "enter is null." );
726  ret= HsmStateError();
727  }
728  }
729  break;
730  case _hsm_exit_raw: {
731  const RawActionEvent* event= (const RawActionEvent*)status->evt;
732  if (!current->desc.exit && event->action) {
733  current->desc.exit= event->action;
734  ret= BuildingBody();
735  }
736  else {
737  Builder_Error( builder, current->desc.exit ? "exit already specified." : "exit is null." );
738  ret= HsmStateError();
739  }
740  }
741  break;
742  case _hsm_exit_ud: {
743  const ActionEvent* event= (const ActionEvent*)status->evt;
744  if (!current->desc.exit && event->action) {
745  current->desc.exit= RunGenericExit;
746  current->exit= event->action;
747  current->exit_ud= event->action_data;
748  ret= BuildingBody();
749  }
750  else {
751  Builder_Error( builder, current->desc.exit ? "exit already specified." : "exit is null." );
752  ret= HsmStateError();
753  }
754  }
755  break;
756  case _hsm_process_raw: {
757  const RawProcessEvent* event= (const RawProcessEvent*)status->evt;
758  if (event->process) {
759  NewProcessRaw( current, event->process );
760  ret= BuildingBody();
761  }
762  else {
763  Builder_Error( builder, "process is null." );
764  ret= HsmStateError();
765  }
766  }
767  break;
768  case _hsm_process_ud: {
769  const ProcessEventUd* event= (const ProcessEventUd*)status->evt;
770  if (event->process) {
771  NewProcessUD( current, event->process, event->process_data );
772  ret= BuildingBody();
773  }
774  else {
775  Builder_Error( builder, "process is null." );
776  ret= HsmStateError();
777  }
778  }
779  break;
780  // start building a event handler
781  case _hsm_guard_raw:
782  case _hsm_guard_ud:
783  {
784  ret= BuildingHandler();
785  }
786  break;
787  case _hsm_end: {
788  // setup the event handler, this also a key that the state is good to go.
789  current->desc.process= RunGenericEvent;
790 
791  // if this was the last matching begin/end pair, we're done
792  // ( we dont use !current, b/c of potential containment in external states )
793  if (--builder->count==0) {
794  builder->current= 0;
795  ret= Building();
796  }
797  else {
798  builder->current= (state_t*) current->desc.parent;
799  ret= BuildingBody();
800  }
801  }
802  break;
803  }
804  return ret;
805 }

852  handler_t* handler= (handler_t*) state->process;
853  HSM_ASSERT( state->process->flags & ProcessHandler );
854 
855  switch (status->evt->type) {
856  case _hsm_goto: {
857  const StateEvent* event= (const StateEvent*)status->evt;
858  const int go= event->id;
859  hash_entry_t* target= Hash_FindEntry( &(builder->hash), go );
860  if (!handler->target && target) {
861  handler->target= target;
862  ret= HsmStateHandled();
863  }
864  else {
865  Builder_Error( builder, handler->target ? "goto already specified" : "unknown goto target" );
866  ret= HsmStateError();
867  }
868  }
869  break;
870  case _hsm_guard_raw:
871  case _hsm_guard_ud:
872  {
873  const GuardEvent* guard_event= (const GuardEvent*)status->evt;
874  // on append, we want to add this guard to us
875  // otherwise we let the parent state start a new event handler
876  if (guard_event->append) {
877  guard_t* guard= NewGuardFromEvent( status, handler );
878  if (guard) {
879  ret= HsmStateHandled();
880  }
881  else {
882  Builder_Error( builder, "couldnt create guard");
883  ret= HsmStateError();
884  }
885  }
886  }
887  break;
888  case _hsm_action_ud: {
889  const ActionEvent* action_event= (const ActionEvent*)status->evt;
890  if (NewAction( handler, action_event->action, action_event->action_data )) {
891  ret= HsmStateHandled();
892  }
893  else {
894  Builder_Error( builder, !action_event->action ? "no action specified" : "couldnt allocate action");
895  ret= HsmStateError();
896  }
897  }
898  break;
899  };
900  }
901  return ret;
902 }
