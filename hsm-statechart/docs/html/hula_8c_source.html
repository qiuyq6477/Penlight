99 static int HulaGetStateTables( lua_State * L )
100 {
101  static int tablespot=0;
102  lua_pushlightuserdata( L, &tablespot ); // unique spot for tables in the registry
103  lua_gettable( L, LUA_REGISTRYINDEX ); // pull registry[tablespot]
104 
105  // is this the first time we're using registry[tablespot]?
106  if (lua_isnil( L, -1 )) {
107  lua_pop(L,1);
108  // create a new table
109  lua_pushlightuserdata( L, &tablespot ); // key,
110  lua_newtable( L ); // value,
111  // store it in the registry
112  lua_settable( L, LUA_REGISTRYINDEX ); // registry[key]=value
113  // pull it back from the registry
114  lua_pushlightuserdata( L, &tablespot );
115  lua_gettable( L, LUA_REGISTRYINDEX );
116  }
117 
118  return lua_gettop( L );
119 }

248 static hsm_context HulaEnterUD( hsm_status status, void * user_data )
249 {
250  hula_context_t* new_ctx=0, *parent_ctx=0;
251  lua_State* L= (lua_State*)user_data;
252  const int event_table= lua_gettop(L);
253 
254  // get the lua specified enter= function()
255  const int lua_entryfn= HulaGetEvent( L, status->state, LUA_T_ENTER, 0 );
256 
257  // get our parent's hula_context_t via ugly magic.
258  // this allows us to pass our lua state, and context data to descendents
259  // maybe there's a better way.....
260  if (!status->state->parent) {
261  if (status->hsm->flags & HSM_FLAGS_HULA) {
262  hula_machine_t* parent= (hula_machine_t*) status->hsm;
263  parent_ctx= &parent->ctx;
264  }
265  }
266  else
267  if (status->state->parent->exit==HulaExit) {
268  parent_ctx= (hula_context_t*) status->ctx;
269  }
270 
271  // if our parent has lua data, use that as this state's initial data; if not: use nil
272  // note: push *before* determining whether the user has specified an entry function;
273  // this data gets used for creating a lua context for this state, regardless.
274  if (!parent_ctx) {
275  lua_pushnil( L );
276  }
277  else {
278  lua_rawgeti( L, LUA_REGISTRYINDEX, parent_ctx->lua_ref );
279  }
280 
281  // no entry function means no unique data for this state:
282  // re-use our parent's context ( if its not null )
283  if (!lua_isfunction( L, lua_entryfn )) {
284  lua_remove( L, lua_entryfn );
285  if (parent_ctx) {
286  lua_pop(L,1);
287  new_ctx= parent_ctx;
288  ++parent_ctx->lua_ref_count;
289  }
290  }
291  else {
292  int err= lua_unpack_and_pcall( L, (status->hsm->flags & HSM_FLAGS_HULA), event_table, HULA_EVENT_NAME, 1 );
293  if (err) {
294  const char * msg=lua_tostring(L,-1);
295  lua_pop(L,1);//? TODO: and do what on error exactly?
296  }
297  else {
298  // provide a "shortcut" so the lua entry handler doesnt have to remember to return the parent ctx:
299  // if they return nil, then we just replace it for them.
300  if (lua_isnil(L,-1)&& (!parent_ctx)) {
301  lua_pop(L,-1);
302  lua_rawgeti( L, LUA_REGISTRYINDEX, parent_ctx->lua_ref );
303  }
304  }
305  }
306 
307  // if we don't have a context container by now, we need one.
308  // it will store a lua_ref to the user's data.
309  if (!new_ctx) {
310  new_ctx= HulaCreateContext( L, -1 );
311  HSM_ASSERT( new_ctx );
312  }
313 
314  HSM_ASSERT( event_table== lua_gettop(L) ); // is life good?
315 
316  // the machine keeps the context
317  return new_ctx ? &(new_ctx->core) : 0;
318 }

322 {
323  int scan;
324  hsm_bool match=HSM_TRUE;
325  // the empty string would mean you match everything
326  for (scan=0; spec[scan]; ++scan) {
327  // but if you run out of test first, you've overspecified
328  if (!test[scan] || (test[scan] != spec[scan])) {
329  match= HSM_FALSE;
330  break;
331  }
332  }

336  match= HSM_FALSE;
337  }

349 static hsm_state HulaRunUD( hsm_status status, void * user_data )
350 {
351  hsm_state ret=0;
352  // context for these states is always a hula context because of on enter
353  hula_context_t*ctx= (hula_context_t*)(status->ctx);
354  HSM_ASSERT( ctx );
355  if (ctx) {
356  lua_State* L= ctx->L;
357  const int event_table= lua_gettop(L);
358  const char * eventspec= (const char*) user_data;
359 
360  // is this the event that's being processed one we care about?
361  hsm_bool matches= HSM_FALSE;
362  hula_callback_is_event cb= HulaGetIsEvent( L );
363  if (cb) {
364  matches= cb( L, eventspec, status->evt );
365  }
366  else {
367  // get the event name from the event table
368  const char * trigger;
369  lua_rawgeti( L, event_table, HULA_EVENT_NAME );
370  trigger= luaL_checkstring( L, -1 );
371  matches= HulaMatchEvent( eventspec, trigger );
372  lua_pop( L, 1 );
373  }
374 
375  if (matches){
376  // push the relevant entry from the state table
377  const int evthandler= HulaGetEvent( L, status->state, 0, eventspec );
378  if (!lua_isfunction( L, evthandler )) {
379  const char * targetname= luaL_checkstring(L, evthandler );
380  ret= targetname ? hsmResolve( targetname ) : 0;
381  if (!ret) {
382  ret= HsmStateError();
383  }
384  lua_pop(L,1); // pop the table entry
385  }
386  else {
387  int err;
388  // context first
389  lua_rawgeti( L, LUA_REGISTRYINDEX, ctx->lua_ref );
390  // unpack the event table, skipping the event name
391  err= lua_unpack_and_pcall( L, (status->hsm->flags & HSM_FLAGS_HULA), event_table, HULA_EVENT_PAYLOAD, 1 );
392  if (err) {
393  const char * msg=lua_tostring(L,-1);
394  lua_pop(L,1);//? TODO: and do what on error exactly?
395  ret= HsmStateError();
396  }
397  else {
398  // evaluate the results
399  if (lua_isstring( L, -1 )) {
400  const char * name= lua_tostring( L, -1 );
401  ret= hsmResolve( name );
402  }
403  else
404  if (lua_isboolean( L, -1 ) && lua_toboolean( L, -1 )) {
405  ret= HsmStateHandled();
406  }
407  // pop the function results
408  lua_pop(L,1);
409  }
410  }
411  HSM_ASSERT( event_table== lua_gettop(L) ); // is life good?
412  }
413  }
414  // return the next state
415  return ret;
416 }

422 static void HulaExit( hsm_status status )
423 {
424  hula_context_t*ctx= (hula_context_t*)(status->ctx);
425  lua_State* L= ctx->L;
426  const int event_table= lua_gettop(L);
427 
428  // pull the function to call:
429  const int lua_exitfn= HulaGetEvent( L, status->state, LUA_T_EXIT, 0 );
430 
431  // call the function: one arg, zero results
432  if (!lua_isfunction( L, lua_exitfn )) {
433  lua_remove( L, lua_exitfn );
434  }
435  else {
436  int err;
437  // get the context
438  lua_rawgeti( L, LUA_REGISTRYINDEX, ctx->lua_ref );
439  // unpack the event object and call the already pushed function

441  if (err) {

443  lua_pop(L,1);//? TODO: and do what on error exactly?
444  }
445  }
446

448  if (--ctx->lua_ref_count <0) {
449  luaL_unref( L, LUA_REGISTRYINDEX, ctx->lua_ref );

452  HSM_ASSERT( event_table== lua_gettop(L) ); // is life good?
453 }

469 static hula_error _HulaBuildState( lua_State*L, const int table, nstring_t statename )
470 {
471  hula_error err= 0;
472  if (!NSTRING(statename)) {
473  err= HULA_ERR_ARG;
474  }
475  else {
476  int initState=0;
477  // check for an 'init' entry
478  // ie. the right side of { -init='s1' }
479  lua_getfield( L, table, INIT );
480  // its okay not to have an init;
481  // ( TODO: a check to make the state deosnt have children? )
482  if (lua_isnil( L, -1 )) {
483  lua_pop(L,1);
484  }
485  else
486  if (!lua_isstring( L, -1) ) {
487  err= HULA_ERR_UNEXPECTED_KEY;
488  lua_pop(L,1);
489  }
490  else {
491  nstring_t initname;
492  lua_pushvalue( L, -1 ); // dupe the name: we need to keep a pointer to it
493  lua_tonstring( L, -2, &initname );
494  // from the main, get table the definition of the init state,
495  // ie. the right side of { 's1'= {} }
496  lua_gettable( L, table );
497  if (!(initState= hsmBegin( initname.string, initname.len ))) {
498  err= "_HulaBuildState: hsmBegin initial state";
499  }
500  else {
501  const int check= lua_gettop(L);
502  err= _HulaBuildState( L, lua_gettop(L), initname );
503  hsmEnd();
504  HSM_ASSERT( check == lua_gettop(L) );
505  }
506  lua_pop( L, 2 ); // pop the init dupe, and the init state's table.
507  }
508 
509  if (!err) {
510  // create a lua table to hold functions
511  const int state_table= HulaCreateStateTable( L, statename );
512 
513  // force each and every lua function to have the context management it needs
514  hsmOnEnterUD( HulaEnterUD, L );
515  hsmOnExit( HulaExit );
516 
517  // walk the contents of the source table
518  lua_pushnil(L);
519  while (lua_next(L, table)) {
520  nstring_t keyname;
521  const int value_idx= lua_gettop(L);
522  const int key_idx = value_idx-1;
523 
524  // we expect that key is always a string. not just stringable.
525  // ( note: in place coercion of non-string key can corrupt lua_next() )
526  if (err || !lua_isstring( L, key_idx )) {
527  err= err ? err : HULA_ERR_UNEXPECTED_KEY;
528  lua_pop(L,2); // pop loop iterators
529  break;
530  }
531 
532  lua_tonstring( L, key_idx, &keyname );
533  // if the key:value is a a table, its a state, we need to create it.
534  if ( lua_istable( L, value_idx ) ) {
535  // but: if the nextState is our initial state...
536  // then we've already created it. ( i wish there was a simpler way )
537  const int nextState= hsmState( keyname.string );
538  if (nextState!= initState) {
539  if (!hsmBegin( keyname.string, keyname.len )) {
540  err= "_HualBuildState:hsmBegin staet";
541  }
542  else {
543  const int check= lua_gettop(L);
544  err= _HulaBuildState( L, value_idx, keyname );
545  HSM_ASSERT( check== lua_gettop(L) );
546  hsmEnd();
547  }
548  }
549  HSM_ASSERT( value_idx == lua_gettop(L) );
550  lua_pop(L,1); // pop `value`, leaving key on top for lua_next() loop
551  }
552  else {
553  const int is_target_function= lua_isfunction( L, value_idx );
554  const int is_target_name= lua_isstring( L, value_idx );
555  if (!is_target_name && !is_target_function) {
556  err= HULA_ERR_UNEXPECTED_VALUE;
557  lua_pop(L,2); // pop loop iterators
558  break;
559  }
560  else {
561  // Init: ex. { init = 's11' } but .... we took care of it already.
562  if (NSTRING_IS( keyname, INIT )) {
563  lua_pop(L,1); // manually pop the value
564  }
565  else
566  // Entry: ex. { enter = function() end }
567  if (is_target_function && NSTRING_IS( keyname, ENTRY )) {
568  // state_table['exit']= function
569  lua_rawseti( L, state_table, LUA_T_ENTER ); // value is popped.
570  }
571  else
572  // Exit: ex. { enter = function() end }
573  if (is_target_function && NSTRING_IS( keyname, EXIT )) {
574  // state_table['exit']= function
575  lua_rawseti( L, state_table, LUA_T_EXIT ); // value is popped.
576  }
577  // Event: ex. { event = 'name' }, or: { event = function() end }
578  else {
579  const char *eventspec= keyname.string;
580  hsmOnEventUD( HulaRunUD, (void*) eventspec );
581  // store state_table[ 'eventspec' ]= target.
582  // to ensure HulaUserIsEvent has a valid 'eventspec' pointer.
583  // not officially supported, but sharing string memory works.
584  // alternative: copy or ref string memory, and remember to clean it up.
585  // might want a named events feature in builder, but avoiding the api complication for now.
586  lua_setfield( L, state_table, eventspec ); // value is popped.
587  }
588  }
589  }
590  }
591  lua_remove( L, state_table );
592  }
593  }
594  return err;
595 }

601 hula_error HulaBuildState( lua_State*L, int chartidx, int * pid )
602 {
603  hula_error err= "HulaBuildState: error";
604  const int check= lua_gettop(L);
605 
606  // we want to get the one and only entry, but we dont know its index
607  lua_pushnil(L);
608 
609  // note: there's nothing wrong, technically, with an empty state
610  if (lua_next(L, chartidx)) { // pops top; pulls key,value on to the stack
611  const int bodyidx = lua_gettop(L);
612  const int nameidx = bodyidx-1;
613  lua_pushvalue( L, nameidx );
614  if (lua_next(L, chartidx )) {
615  err= "only state={body} pair can be in the top state";
616  lua_pop(L,4); // pop both sets of iterators
617  }
618  else {
619  if (!lua_isstring( L, nameidx )) {
620  err= "all keys must be strings";
621  }
622  else
623  if (!lua_istable( L, bodyidx )) {
624  err= "expected a state object";
625  }
626  else {
627  nstring_t name;
628  lua_tonstring( L, nameidx, &name );
629  err= HulaBuildNamedState( L, bodyidx, name.string, name.len, pid );
630  }
631  lua_pop(L,2); // pop iterators
632  }
633  }
634 
635  HSM_ASSERT( check == lua_gettop(L) );
636  return err;
637 }
